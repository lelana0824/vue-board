<template>
  <div>
    <form>
      <input v-model="writer" placeholder="글쓴이"/>
      <input v-model="title" placeholder="제목"/>
      <textarea v-model="content" placeholder="내용"/>
    </form>
    <button @click="create">확인</button>
  </div>
</template>
<script>
import {data} from '@/data'
 
export default {
  name: 'Create',
  data() {
    return {
      writer: '',
      title: '',
      content: ''
    }
  },
  methods: {
    create() {
      data.push({
        writer: this.writer,
        title: this.title,
        content: this.content
      })
      this.$router.push({
        path: '/'
      })
    }
  }
}
</script>
